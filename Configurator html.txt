  <script>
    const data = [
      { platform: 'Lexis+', tier: 'State Essential', price: '$100', features: 'Basic search; Case law', logo: 'LexisPlus.png' },
      { platform: 'Lexis+', tier: 'State Enhanced', price: '$150', features: 'Enhanced search; Case law; Statutes', logo: 'LexisPlus.png' },
      { platform: 'Lexis+', tier: 'State Professional', price: '$200', features: 'All Enhanced + Analytics', logo: 'LexisPlus.png' },
      { platform: 'Lexis+', tier: 'Premiere', price: '$250', features: 'All Professional + AI Summaries', logo: 'LexisPlus.png' },
      { platform: 'Lexis+ AI', tier: 'State Essential', price: '$120', features: 'AI Assist Basic', logo: 'LexisPlusAi.png' },
      { platform: 'Lexis+ AI', tier: 'State Enhanced', price: '$180', features: 'AI Enhanced Search; Summaries', logo: 'LexisPlusAi.png' },
      { platform: 'Lexis+ AI', tier: 'State Professional', price: '$240', features: 'All Enhanced + Advanced AI', logo: 'LexisPlusAi.png' },
      { platform: 'Lexis+ AI', tier: 'Premiere', price: '$300', features: 'Full AI Suite + Analytics', logo: 'LexisPlusAi.png' }
    ];

    const pSelect = document.getElementById('platform');
    const tSelect = document.getElementById('tier');
    const logoImg  = document.getElementById('logo');
    const titleEl  = document.getElementById('title');
    const priceEl  = document.querySelector('#output p strong + span, #output p:nth-child(3)') || document.createElement('span');
    const featuresEl = document.getElementById('features');

    function updateCard() {
      const plat = pSelect.value;
      const tier = tSelect.value;
      const entry = data.find(d => d.platform === plat && d.tier === tier);
      if (!entry) return;
      logoImg.src = entry.logo;
      titleEl.textContent = `${plat} - ${tier}`;
      priceEl.parentNode.innerHTML = `<strong>Price:</strong> ${entry.price}`;
      featuresEl.innerHTML = `<strong>Features:</strong> ${entry.features}`;
    }

    pSelect.addEventListener('change', updateCard);
    tSelect.addEventListener('change', updateCard);

    updateCard();
  </script>
